fmt = "%Y-%m-%d %H:%M:%S"
def getHour(row):
    thisday = datetime.strptime(row['datetime'], fmt)
    return thisday.hour

def getMonth(row):
    thisday = datetime.strptime(row['datetime'], fmt)
    return thisday.month

def getDay(row):
    thisday = datetime.strptime(row['datetime'], fmt)
    return thisday.weekday()

def getYear(row):
    thisday = datetime.strptime(row['datetime'], fmt)
    return thisday.year
